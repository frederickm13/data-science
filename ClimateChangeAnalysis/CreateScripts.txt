CREATE TABLE CITIES (
    ID SERIAL PRIMARY KEY,
    ACCOUNT_NUMBER INT,
    ORGANIZATION VARCHAR(200),
    CITY VARCHAR(100),
    COUNTRY VARCHAR(100),
    CDP_REGION VARCHAR(100),
    REPORTING_AUTHORITY VARCHAR(200),
    ACCESS VARCHAR(25),
    LOCATION VARCHAR(100)
);

CREATE TABLE POPULATIONS (
    ID SERIAL PRIMARY KEY,
    CITY_ID INT REFERENCES CITIES(ID),
    POPULATION_YEAR SMALLINT,
    POPULATION INT
);

CREATE TABLE CITY_REPORTING_YEARS (
    ID SERIAL PRIMARY KEY,
    CITY_ID INT REFERENCES CITIES(ID),
    REPORTING_YEAR SMALLINT
);

CREATE TABLE CORPORATIONS (
    ID SERIAL PRIMARY KEY,
    ACCOUNT_NUMBER INT,
    ORGANIZATION VARCHAR(200),
    COUNTRY VARCHAR(100),
    REGION VARCHAR(50),
    PUBLIC VARCHAR(25),
    AUTHORITY_TYPES VARCHAR(50),
    ACTIVITIES VARCHAR(300),
    SECTORS VARCHAR(300),
    INDUSTRIES VARCHAR(200),
    PRIMARY_ACTIVITY VARCHAR(100),
    PRIMARY_SECTOR VARCHAR(100),
    PRIMARY_INDUSTRY VARCHAR(100),
    PRIMARY_TICKER VARCHAR(25),
    TICKERS VARCHAR(100)
);

CREATE TABLE CORPORATIONS_REPORTING_YEARS (
    ID SERIAL PRIMARY KEY,
    CORP_ID INT REFERENCES CORPORATIONS(ID),
    REPORTING_YEAR SMALLINT,
    SAMPLES VARCHAR(300),
    RESPONSE_RECEIVED_DATE TIMESTAMP,
    MINIMUM_TIER VARCHAR(100),
    SELECTED_TIER VARCHAR(100),
    QUESTIONNAIRE VARCHAR(50),
    THEME VARCHAR(50),
    PRIMARY_QUESTIONNAIRE_SECTOR VARCHAR(50)
);

CREATE TABLE CITY_QUESTIONS (
    ID SERIAL PRIMARY KEY,
    CITY_ID INT REFERENCES CITIES(ID),
    QUESTIONNARE VARCHAR(25),
    SURVEY_YEAR SMALLINT,
    PARENT_SECTION VARCHAR(100),
    SECTION VARCHAR(100),
    QUESTION_NUMBER VARCHAR(10),
    QUESTION_NAME VARCHAR(500),
    COLUMN_NUMBER SMALLINT,
    COLUMN_NAME VARCHAR(300),
    ROW_NUMBER SMALLINT,
    ROW_NAME VARCHAR(300)
);

CREATE TABLE CITY_RESPONSES (
    ID SERIAL PRIMARY KEY,
    CITY_ID INT REFERENCES CITIES(ID),
    QUESTION_ID INT REFERENCES CITY_QUESTIONS(ID),
    SURVEY_YEAR SMALLINT,
    RESPONSE VARCHAR(1000),
    COMMENTS VARCHAR(1000)
);

CREATE TABLE CORP_QUESTIONS (
    ID SERIAL PRIMARY KEY,
    CORP_ID INT REFERENCES CORPORATIONS(ID),
    SURVEY_YEAR SMALLINT,
    RESPONSE_RECEIVED_DATE TIMESTAMP,
    SUBMISSION_DATE TIMESTAMP,
    ACCOUNTING_PERIOD_TO VARCHAR(25),
    ORS_RESPONSE_ID INT,
    PAGE_NAME VARCHAR(25),
    MODULE_NAME VARCHAR(50),
    QUESTION_NUMBER VARCHAR(10),
    QUESTION_NAME VARCHAR(500),
    COLUMN_NUMBER SMALLINT,
    COLUMN_NAME VARCHAR(300),
    ROW_NUMBER SMALLINT,
    ROW_NAME VARCHAR(300),
    DATA_POINT_ID INT
);

CREATE TABLE CORP_RESPONSES (
    ID SERIAL PRIMARY KEY,
    CORP_ID INT REFERENCES CORPORATIONS(ID),
    QUESTION_ID INT REFERENCES CORP_QUESTIONS(ID),
    SURVEY_YEAR SMALLINT,
    DATA_POINT_ID INT,
    RESPONSE VARCHAR(1000),
    COMMENTS VARCHAR(1000)
);
